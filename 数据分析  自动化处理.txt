"""
生信数据分析自动化处理系统
功能：
1. 模拟/读取表达矩阵数据
2. 自动差异基因筛选
3. 富集分析简易模拟
4. 自动绘制火山图、热图
5. 自动生成数据分析报告
6. 全流程无人工干预自动化
"""
import os
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from scipy import stats

# ===================== 全局配置 =====================
plt.rcParams["font.sans-serif"] = ["SimHei"]
plt.rcParams["axes.unicode_minus"] = False
# 结果保存文件夹
OUT_DIR = "bioinfo_auto_result"
os.makedirs(OUT_DIR, exist_ok=True)

# ===================== Agent1：数据生成/读取 =====================
def load_or_generate_data(n_genes=1000, n_sample=10):
    """
    自动生成模拟基因表达矩阵，也可替换为读取本地csv
    """
    np.random.seed(12345)
    # 基因名
    genes = [f"Gene_{i:04d}" for i in range(n_genes)]
    # 对照组、实验组表达
    ctrl = np.random.normal(loc=5, scale=1.2, size=(n_genes, n_sample))
    treat = np.random.normal(loc=5.5, scale=1.2, size=(n_genes, n_sample))
    
    # 构造差异基因
    diff_idx = np.random.choice(n_genes, size=150, replace=False)
    ctrl[diff_idx[:75]] -= 1.8
    treat[diff_idx[75:]] += 1.8

    # 拼接矩阵
    expr = pd.DataFrame(
        np.hstack([ctrl, treat]),
        index=genes,
        columns=[f"Ctrl_{i+1}" for i in range(n_sample)] + 
                [f"Treat_{i+1}" for i in range(n_sample)]
    )
    expr.to_csv(os.path.join(OUT_DIR, "gene_expression_matrix.csv"))
    print("[数据处理Agent] 表达矩阵生成完成，已保存")
    return expr

# ===================== Agent2：自动差异分析 =====================
def diff_analysis(expr):
    """
    自动t检验 + 筛选差异基因 |log2FC|>1 & P<0.05
    """
    ctrl_cols = [c for c in expr.columns if "Ctrl" in c]
    treat_cols = [c for c in expr.columns if "Treat" in c]
    
    res_list = []
    for gene in expr.index:
        ctrl_val = expr.loc[gene, ctrl_cols].values
        treat_val = expr.loc[gene, treat_cols].values
        
        # t检验
        t_stat, p_val = stats.ttest_ind(treat_val, ctrl_val)
        # 计算log2FC
        mean_ctrl = np.mean(ctrl_val)
        mean_treat = np.mean(treat_val)
        log2fc = np.log2(mean_treat / mean_ctrl) if mean_ctrl != 0 else 0
        
        res_list.append({
            "Gene": gene,
            "log2FC": log2fc,
            "Pvalue": p_val
        })
    
    diff_res = pd.DataFrame(res_list)
    # 差异筛选标准
    diff_res["Regulate"] = "NoDiff"
    diff_res.loc[(diff_res["log2FC"] > 1) & (diff_res["Pvalue"] < 0.05), "Regulate"] = "Up"
    diff_res.loc[(diff_res["log2FC"] < -1) & (diff_res["Pvalue"] < 0.05), "Regulate"] = "Down"
    
    diff_res.to_csv(os.path.join(OUT_DIR, "diff_gene_result.csv"), index=False)
    print("[差异分析Agent] 差异基因分析完成，上调/下调/无差异自动分类")
    return diff_res

# ===================== Agent3：自动可视化 火山图 =====================
def plot_volcano(diff_res):
    plt.figure(figsize=(10, 6))
    sns.scatterplot(
        data=diff_res, x="log2FC", y=-np.log10(diff_res["Pvalue"]),
        hue="Regulate", palette={"Up":"#e74c3c", "Down":"#3498db", "NoDiff":"#95a5a6"},
        s=15, alpha=0.8
    )
    plt.axvline(x=1, linestyle="--", color="gray")
    plt.axvline(x=-1, linestyle="--", color="gray")
    plt.title("差异基因火山图", fontsize=14)
    plt.tight_layout()
    plt.savefig(os.path.join(OUT_DIR, "volcano_plot.png"), dpi=300, bbox_inches="tight")
    plt.close()
    print("[可视化Agent] 火山图绘制完成")

# ===================== Agent4：自动简易富集分析 =====================
def simple_enrichment(diff_res):
    """
    自动统计上下调基因数量 + 模拟通路富集结果
    """
    up_num = sum(diff_res["Regulate"] == "Up")
    down_num = sum(diff_res["Regulate"] == "Down")
    no_diff_num = sum(diff_res["Regulate"] == "NoDiff")

    enrich_text = f"""
=== 自动化富集分析结果 ===
总基因数：{len(diff_res)}
上调基因数：{up_num}
下调基因数：{down_num}
无差异基因数：{no_diff_num}

富集通路Top5（模拟）：
1. 细胞纤维化调控通路
2. 免疫浸润相关信号通路
3. 细胞凋亡与自噬通路
4. 炎症反应调控通路
5. 代谢重编程通路
"""
    with open(os.path.join(OUT_DIR, "enrichment_report.txt"), "w", encoding="utf-8") as f:
        f.write(enrich_text)
    print("[富集分析Agent] 自动统计与富集报告生成完成")
    return enrich_text

# ===================== Agent5：自动生成完整数据分析报告 =====================
def generate_final_report(diff_res, enrich_text):
    up_num = sum(diff_res["Regulate"] == "Up")
    down_num = sum(diff_res["Regulate"] == "Down")
    report = f"""
# 生物信息学数据分析自动化报告
## 1. 流程说明
本系统实现基因表达数据从**数据导入→差异分析→可视化→富集解读→报告输出**全流程自动化，无需人工逐步操作。

## 2. 分析参数
差异筛选标准：|log2FC| > 1 且 Pvalue < 0.05

## 3. 结果统计
- 检测基因总数：{len(diff_res)}
- 显著上调基因：{up_num} 个
- 显著下调基因：{down_num} 个

## 4. 富集分析概要
{enrich_text}

## 5. 系统价值
将传统需要3~5天的生信数据分析流程，压缩至几分钟完成，标准化流程、避免人为操作误差，自动输出图表与可直接用于论文的分析报告。

    with open(os.path.join(OUT_DIR, "final_analysis_report.md"), "w", encoding="utf-8") as f:

# ===================== 主流程：全自动化流水线 =====================
if __name__ == "__main__":
